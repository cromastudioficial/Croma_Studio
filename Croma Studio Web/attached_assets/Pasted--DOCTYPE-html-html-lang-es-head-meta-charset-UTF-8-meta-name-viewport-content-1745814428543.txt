                const size = Math.random() * 10 + 5;

        });

        increaseQty.addEventListener('click', () => {
            quantityInput.value++;
        });

        // Agregar productos al carrito (abre modal de personalización)
        document.querySelectorAll('.add-to-cart').forEach(button => {
            button.addEventListener('click', function() {
                currentProduct = {
                    id: this.getAttribute('data-id'),
                    name: this.getAttribute('data-name'),
                    price: parseFloat(this.getAttribute('data-price')),
                    image: this.getAttribute('data-image')
                };
                
                customizationModal.style.display = 'flex';
                overlay.classList.add('active');
                
                // Resetear modal
                designPreview.src = '';
                designPreview.style.display = 'none';
                quantityInput.value = 1;
                specificationsInput.value = '';
                designInput.value = '';
                uploadedFile = null;
            });
        });

        // Confirmar personalización y agregar al carrito
        addToCartFinal.addEventListener('click', () => {
            if (!currentProduct) return;
            
            const selectedSize = document.querySelector('.size-option.selected').textContent;
            const quantity = parseInt(quantityInput.value);
            const specifications = specificationsInput.value;
            
            // Verificar si el producto ya está en el carrito
            const existingItem = cart.find(item => 
                item.id === currentProduct.id && 
                item.size === selectedSize && 
                item.specifications === specifications
            );
            
            if (existingItem) {
                existingItem.quantity += quantity;
            } else {
                const cartItem = {
                    ...currentProduct,
                    size: selectedSize,
                    design: designPreview.src || null,
                    specifications: specifications,
                    quantity: quantity,
                    file: uploadedFile
                };
                
                cart.push(cartItem);
            }
            
            updateCart();
            
            // Cerrar modal
            customizationModal.style.display = 'none';
            overlay.classList.remove('active');
            
            // Mostrar el carrito
            cartSidebar.classList.add('active');
            overlay.classList.add('active');
        });

        // Actualizar el carrito
        function updateCart() {
            // Actualizar contador
            const totalItems = cart.reduce((sum, item) => sum + item.quantity, 0);
            document.querySelector('.cart-count').textContent = totalItems;
            
            // Actualizar lista de productos
            if (cart.length === 0) {
                emptyCartMessage.style.display = 'block';
                cartItemsContainer.innerHTML = '<p class="empty-cart-message">Tu carrito está vacío</p>';
            } else {
                emptyCartMessage.style.display = 'none';
                cartItemsContainer.innerHTML = '';
                
                cart.forEach(item => {
                    const cartItemElement = document.createElement('div');
                    cartItemElement.className = 'cart-item';
                    cartItemElement.innerHTML = `
                        <div class="cart-item-image">
                            <img src="${item.image}" alt="${item.name}">
                        </div>
                        <div class="cart-item-details">
                            <div class="cart-item-title">${item.name} (${item.size})</div>
                            <div class="cart-item-price">$${item.price.toFixed(2)} MXN</div>
                            ${item.specifications ? `<div class="cart-item-specs">Especificaciones: ${item.specifications}</div>` : ''}
                            <div class="cart-item-quantity">
                                <button class="quantity-btn decrease" data-id="${item.id}" data-size="${item.size}" data-specs="${encodeURIComponent(item.specifications || '')}">-</button>
                                <input type="text" class="quantity-input" value="${item.quantity}" readonly>
                                <button class="quantity-btn increase" data-id="${item.id}" data-size="${item.size}" data-specs="${encodeURIComponent(item.specifications || '')}">+</button>
                            </div>
                        </div>
                        <button class="remove-item" data-id="${item.id}" data-size="${item.size}" data-specs="${encodeURIComponent(item.specifications || '')}"><i class="fas fa-trash"></i></button>
                    `;
                    
                    cartItemsContainer.appendChild(cartItemElement);
                });
                
                // Agregar eventos a los botones de cantidad y eliminar
                document.querySelectorAll('.decrease').forEach(button => {
                    button.addEventListener('click', () => {
                        const id = button.getAttribute('data-id');
                        const size = button.getAttribute('data-size');
                        const specs = decodeURIComponent(button.getAttribute('data-specs'));
                        const item = cart.find(item => 
                            item.id === id && 
                            item.size === size && 
                            item.specifications === specs
                        );
                        
                        if (item.quantity > 1) {
                            item.quantity -= 1;
                        } else {
                            cart = cart.filter(item => 
                                !(item.id === id && 
                                item.size === size && 
                                item.specifications === specs)
                            );
                        }
                        
                        updateCart();
                    });
                });
                
                document.querySelectorAll('.increase').forEach(button => {
                    button.addEventListener('click', () => {
                        const id = button.getAttribute('data-id');
                        const size = button.getAttribute('data-size');
                        const specs = decodeURIComponent(button.getAttribute('data-specs'));
                        const item = cart.find(item => 
                            item.id === id && 
                            item.size === size && 
                            item.specifications === specs
                        );
                        item.quantity += 1;
                        updateCart();
                    });
                });
                
                document.querySelectorAll('.remove-item').forEach(button => {
                    button.addEventListener('click', () => {
                        const id = button.getAttribute('data-id');
                        const size = button.getAttribute('data-size');
                        const specs = decodeURIComponent(button.getAttribute('data-specs'));
                        cart = cart.filter(item => 
                            !(item.id === id && 
                            item.size === size && 
                            item.specifications === specs)
                        );
                        updateCart();
                    });
                });
            }
            
            // Actualizar total
            const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
            cartTotalElement.textContent = `$${total.toFixed(2)} MXN`;
        }

        // Botón de checkout
        checkoutBtn.addEventListener('click', async () => {
            if (cart.length === 0) {
                alert('Tu carrito está vacío');
                return;
            }
            
            // Redirigir a WhatsApp con el pedido
            const phoneNumber = '525539153056';
            let message = '¡Hola! Quiero realizar el siguiente pedido:\n\n';
            
            cart.forEach(item => {
                message += `- ${item.name} (Talla: ${item.size}, Cantidad: ${item.quantity} x $${item.price.toFixed(2)})\n`;
                if (item.specifications) {
                    message += `  Especificaciones: ${item.specifications}\n`;
                }
            });
            
            message += `\nTotal: $${cart.reduce((sum, item) => sum + (item.price * item.quantity), 0).toFixed(2)} MXN`;
            
            // Crear URL de WhatsApp
            let whatsappUrl = `https://wa.me/${phoneNumber}?text=${encodeURIComponent(message)}`;
            
            // Si hay imágenes para enviar
            const itemsWithFiles = cart.filter(item => item.file);
            if (itemsWithFiles.length > 0) {
                // Para el propósito de este ejemplo, mostraremos un mensaje
                alert('Por favor, comparte las imágenes de tu diseño directamente en el chat de WhatsApp después de abrir el enlace.');
                
                // En un entorno real, podrías usar la API de WhatsApp Business para enviar archivos
                // o subir las imágenes a un servidor y compartir el enlace
            }
            
            window.open(whatsappUrl, '_blank');
            
            // Vaciar el carrito después del pedido
            cart = [];
            updateCart();
            cartSidebar.classList.remove('active');
            overlay.classList.remove('active');
        });

        // Formulario de contacto
        const contactForm = document.getElementById('contactForm');
        contactForm.addEventListener('submit', function(e) {
            e.preventDefault();
            
            const name = document.getElementById('name').value;
            const email = document.getElementById('email').value;
            const phone = document.getElementById('phone').value;
            const message = document.getElementById('message').value;
            
            // Aquí podrías agregar el código para enviar el formulario
            // Por ejemplo, usando fetch() para enviar los datos a un servidor
            
            alert(`Gracias ${name}, tu mensaje ha sido enviado. Nos pondremos en contacto contigo pronto.`);
            contactForm.reset();
        });

        // Inicializar mapa cuando la página cargue
        window.addEventListener('load', initMap);
    </script>
